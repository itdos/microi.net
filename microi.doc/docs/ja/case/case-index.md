> * Sql文は次のとおりです。成功事例が4文字を超えるごとに4文字目を * 、4文字以下で1文字目を * に置き換えます
```sql
-- SET SESSION group_concat_max_len = 1000000;  -- 增加到1MB
SELECT GROUP_CONCAT(CONCAT('`', CASE WHEN CHAR_LENGTH(ClientName) > 4 
    THEN CONCAT(SUBSTRING(ClientName, 1, 3), '*',  SUBSTRING(ClientName, 5))
    ELSE  CONCAT('*', SUBSTRING(ClientName, 2))
    END, '`') SEPARATOR ' ') AS client_names
FROM ( SELECT DISTINCT ClientName, CreateTime  FROM sys_osclients) AS temp
WHERE ClientName <> '' 
    AND ClientName IS NOT NULL 
    AND ClientName NOT LIKE '%测试%'
    AND ClientName NOT LIKE '%test%' 
    AND ClientName NOT LIKE '%demo%'
ORDER BY CreateTime DESC
```
